<VirtualHost *:80>
    ServerName example.com  # Replace with your domain or IP address

    WSGIDaemonProcess myflaskapp python-home=/var/www/myflaskapp/venv python-path=/var/www/myflaskapp
    WSGIProcessGroup myflaskapp
    WSGIScriptAlias / /var/www/myflaskapp/flaskapp.wsgi

    <Directory /var/www/myflaskapp>
        WSGIProcessGroup myflaskapp
        WSGIApplicationGroup %{GLOBAL}
        Require all granted
    </Directory>

    ErrorLog /var/log/httpd/myflaskapp_error.log
    CustomLog /var/log/httpd/myflaskapp_access.log combined
</VirtualHost>
